2. Next.js Implementation Logic
Page Component (app/v1/[niche]/page.tsx)
The page must parse the search parameters and map them to the JSON data.

TypeScript
export default function DemoPage({ searchParams, params }) {
  const { name, 'color-theme': themeId, typography: typoId, type: typeId } = searchParams;
  const { niche } = params;

  // 1. Find Data
  const activeTheme = themes.find(t => t.id === Number(themeId)) || themes[0];
  const activeTypo = typographies.find(t => t.id === Number(typoId)) || typographies[0];
  const designConfig = references[niche][`type_${typeId}`] || references[niche]["type_1"];

  // 2. Inject CSS Variables
  const styleVariables = {
    '--primary': activeTheme.primary,
    '--secondary': activeTheme.secondary,
    '--accent': activeTheme.accent,
    '--text-main': activeTheme.text,
    '--font-heading': activeTypo.heading,
    '--font-body': activeTypo.body,
    '--radius-pill': '9999px',
    '--radius-card': '1.5rem'
  } as React.CSSProperties;

  return (
    <div style={styleVariables} className="font-body text-text-main bg-secondary min-h-screen">
      {designConfig.layout.map(section => (
        <DynamicSection key={section} type={section} businessName={name} />
      ))}
      <HoldMyTeaWidget businessName={name} />
    </div>
  );
}
